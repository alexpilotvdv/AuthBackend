CREATE DATABASE newdb DEFAULT CHARACTER SET utf8 DEFAULT COLLATE utf8_general_ci;

use newdb
//////
CREATE TABLE users (id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT, name VARCHAR(30), email VARCHAR(30),
       password VARCHAR(100), isactivated CHAR(1) DEFAULT '0', activationlink VARCHAR(50));

CREATE TABLE token (id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT, user INTEGER NOT NULL, 
 refreshtoken VARCHAR(500));
